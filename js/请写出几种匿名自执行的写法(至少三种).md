- 家喻户晓类：

```js
// 家喻户晓
(function() {})();
[(function() {})()];
```

- 运算符驱动类：

```js
~(function(w) {
  alert(w.location.href + "," + 11);
})(window);

!(function(w) {
  alert(w.location.href + "," + 11);
})(window);

+(function(w) {
  alert(w.location.href + "," + 11);
})(window);

-(function(w) {
  alert(w.location.href + "," + 11);
})(window);
```
